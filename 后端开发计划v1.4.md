# 书法成长树小程序 - 后端开发计划（v1.4 生产就绪版）

## 一、项目概述

### 1.1 项目背景
书法成长树是一个书法教育培训机构的管理系统，采用"成长树"的比喻来可视化学生的书法学习进度。系统包含教师端（管理学员、记录积分）和学生端（查看成长树、练习册进度、上传作品）。

### 1.2 技术栈
- **框架**: FastAPI (Python 3.10+)
- **数据库**: PostgreSQL + SQLAlchemy ORM (异步驱动 asyncpg)
- **缓存**: Redis (必选，用于 JWT 黑名单、微信 session_key 缓存)
- **HTTP客户端**: httpx (异步)
- **依赖管理**: Poetry 或 uv
- **部署**: Docker + Gunicorn + Volume 挂载

**关键：改密后 Token 拦截验证**
```python
# api/dependencies.py
from datetime import datetime, timezone

async def get_current_user(
    token: str = Depends(oauth2_scheme),
    redis: Redis = Depends(get_redis),
    db: AsyncSession = Depends(get_db)
) -> User:
    """获取当前用户，并验证 Token 有效性"""
    # 1. 解码 Token
    payload = jwt.decode(token, SECRET_KEY, algorithms=["HS256"])
    jti = payload.get("jti")
    user_id = payload.get("sub")
    token_iat = payload.get("iat")

    # 2. 检查 Redis 黑名单
    if await redis.exists(f"blacklist:{jti}"):
        raise HTTPException(status_code=401, detail="Token已失效")

    # 3. 获取用户（包含 password_changed_at）
    user = await db.get(User, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")

    # 4. 【关键】检查密码是否已修改
    if user.password_changed_at:
        # 将 timestamp 转换为带时区的 datetime
        token_iat_dt = datetime.fromtimestamp(token_iat, tz=timezone.utc)
        if token_iat_dt < user.password_changed_at:
            raise HTTPException(
                status_code=401,
                detail="密码已修改，请重新登录"
            )

    return user
```

```python
# services/auth_service.py
from datetime import datetime, timezone

async def change_password(
    teacher_id: str,
    old_password: str,
    new_password: str,
    db: AsyncSession
):
    """修改密码 - 同时更新 password_changed_at 使旧 Token 失效"""
    teacher = await db.get(Teacher, teacher_id)

    # 验证旧密码
    if not await verify_password(old_password, teacher.password_hash):
        raise HTTPException(status_code=400, detail="旧密码错误")

    # 更新密码和修改时间
    teacher.password_hash = await hash_password(new_password)
    teacher.password_changed_at = datetime.now(timezone.utc)  # 【关键】使所有旧 Token 失效

    await db.commit()

    return {"message": "密码修改成功，请使用新密码重新登录"}
```

```python
# services/wechat_service.py
import httpx
from datetime import datetime, timezone

async def wechat_login(code: str, db: AsyncSession, redis: Redis):
    """
    微信小程序登录 - 缓存 session_key 用于后续手机号解密

    【关键】session_key 有效期与微信一致（建议 30 分钟），必须存入 Redis
    """
    # 1. 请求微信服务器
    async with httpx.AsyncClient() as client:
        response = await client.get(
            "https://api.weixin.qq.com/sns/jscode2session",
            params={
                "appid": settings.WECHAT_APPID,
                "secret": settings.WECHAT_SECRET,
                "js_code": code,
                "grant_type": "authorization_code"
            }
        )
        wx_data = response.json()

    openid = wx_data.get("openid")
    session_key = wx_data.get("session_key")  # 【关键】必须保存
    unionid = wx_data.get("unionid")  # 可能为 None

    if not openid or not session_key:
        raise HTTPException(status_code=400, detail="微信登录失败")

    # 2. 查找或创建学员
    result = await db.execute(
        select(Student).where(Student.openid == openid, Student.is_active == True)
    )
    student = result.scalar_one_or_none()

    is_new_user = False
    if not student:
        is_new_user = True
        student = Student(
            openid=openid,
            unionid=unionid,
            name="微信用户",
            is_active=True,
            created_at=datetime.now(timezone.utc)
        )
        db.add(student)
        await db.commit()
        await db.refresh(student)

    # 3. 【关键】将 session_key 存入 Redis（TTL 30 分钟）
    await redis.setex(
        f"wx_session:{student.id}",
        1800,  # 30 分钟过期
        session_key
    )

    # 4. 生成 JWT
    token = create_jwt({
        "sub": str(student.id),
        "type": "student",
        "iat": datetime.now(timezone.utc)
    })

    return {
        "access_token": token,
        "token_type": "bearer",
        "is_new_user": is_new_user,
        "student": student
    }
```

**关键：微信手机号解密（使用缓存的 session_key）**
```python
# services/wechat_service.py
from Crypto.Cipher import AES
import base64
import json

async def decrypt_phone_number(
    student_id: str,
    encrypted_data: str,
    iv: str,
    redis: Redis
) -> str:
    """
    解密微信手机号

    【前置条件】用户必须先登录，session_key 已存入 Redis
    """
    # 1. 从 Redis 获取 session_key
    session_key = await redis.get(f"wx_session:{student_id}")
    if not session_key:
        raise HTTPException(
            status_code=400,
            detail="session_key 已过期，请重新登录"
        )

    # 2. AES-128-CBC 解密
    try:
        session_key_bytes = base64.b64decode(session_key)
        encrypted_bytes = base64.b64decode(encrypted_data)
        iv_bytes = base64.b64decode(iv)

        cipher = AES.new(session_key_bytes, AES.MODE_CBC, iv_bytes)
        decrypted_bytes = cipher.decrypt(encrypted_bytes)

        # 去除 PKCS7 填充
        pad_len = decrypted_bytes[-1]
        decrypted_bytes = decrypted_bytes[:-pad_len]

        # 解析 JSON
        phone_data = json.loads(decrypted_bytes.decode('utf-8'))
        phone_number = phone_data.get("phoneNumber")

        return phone_number

    except Exception as e:
        raise HTTPException(status_code=400, detail=f"手机号解密失败: {str(e)}")
```

```python
class GrowthStageCalculator:
    STAGES = {
        'sprout': (0, 1499, '萌芽宝宝'),
        'seedling': (1500, 2999, '努力伸腰'),
        'small': (3000, 4499, '撑起小伞'),
        'medium': (4500, 5999, '有模有样'),
        'large': (6000, 7499, '披上绿袍'),
        'xlarge': (7500, 8999, '绿意满满'),
        'fruitful': (9000, float('inf'), '硕果累累'),
    }

    @classmethod
    def calculate_stage(cls, total_score: int) -> str:
        for stage, (min_score, max_score, _) in cls.STAGES.items():
            if min_score <= total_score <= max_score:
                return stage
        return 'sprout'
```

#### 2.4.2 资深学员"只升不降"逻辑
```python
async def update_senior_status(student_id: str, db: AsyncSession):
    """更新资深学员状态 - 只升不降原则"""
    result = await db.execute(select(Student).where(Student.id == student_id))
    student = result.scalar_one()

    is_currently_senior = student.root_score + student.trunk_score >= 4500

    if is_currently_senior and not student.ever_reached_senior:
        await db.execute(
            update(Student)
            .where(Student.id == student_id)
            .values(
                is_senior=True,
                ever_reached_senior=True,
                updated_at=datetime.now(timezone.utc)
            )
        )
        await db.commit()
```

## 五、开发注意事项（v1.4 更新）

### 5.1 新增关键配置

| 配置项 | 说明 | 必需 |
|--------|------|------|
| **CORS_ORIGINS** | Web 端跨域白名单 | 是 |
| **REDIS_PASSWORD** | Redis 连接密码 | 是 |
| **WECHAT_APPID/SECRET** | 微信小程序凭证 | 是 |

### 5.2 关键编码规范

#### CORS 必须在路由之前配置
```python
# ✅ 正确
app = FastAPI()
app.add_middleware(CORSMiddleware, ...)  # 先配置中间件
app.include_router(router)  # 后注册路由

# 访问前端 http://localhost:5173 不会报 CORS 错误
```

#### session_key 必须存入 Redis
```python
# ✅ 正确 - 登录时存入 Redis
await redis.setex(f"wx_session:{student_id}", 1800, session_key)

# ✅ 正确 - 解密时取出使用
session_key = await redis.get(f"wx_session:{student_id}")
```

#### 改密后旧 Token 自动失效
```python
# ✅ 正确 - 修改密码时更新时间戳
teacher.password_changed_at = datetime.now(timezone.utc)

# ✅ 正确 - 验证时对比时间戳
if token_iat < user.password_changed_at:
    raise HTTPException(status_code=401, detail="密码已修改")
```

#### Alembic 必须使用异步模板
```bash
# ✅ 正确
alembic init -t async alembic

# ❌ 错误 - 会导致 asyncpg 错误
alembic init alembic
```

## 八、风险与应对（v1.4 更新）

| 风险 | 影响 | 应对措施 |
|------|------|----------|
| **CORS 跨域错误** | **高** | **配置 CORS_ORIGINS 环境变量，确保中间件在所有路由之前** |
| **session_key 丢失导致手机号解密失败** | **高** | **登录时必须存入 Redis，解密时取出，设置 30 分钟 TTL** |
| **Alembic 异步配置错误** | **高** | **使用 `alembic init -t async alembic` 初始化** |
| **改密后旧 Token 仍有效** | **中** | **验证时对比 token_iat 和 password_changed_at** |
| 微信登录集成复杂 | 中 | 提前申请微信小程序，预留联调时间 |
| 并发积分操作 | 高 | 使用悲观锁或原子更新 |
